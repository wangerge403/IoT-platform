 基于 Vue 3 + Element Plus，登录功能
```java
// 在 com.ny.config 包下创建 CorsConfig.java
package com.ny.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.UrlBasedCorsConfigurationSource;
import org.springframework.web.filter.CorsFilter;

@Configuration
public class CorsConfig {
    @Bean
    public CorsFilter corsFilter() {
        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
        CorsConfiguration config = new CorsConfiguration();
        // 允许前端域名访问（* 表示允许所有，生产环境建议指定具体域名）
        config.addAllowedOriginPattern("*");
        // 允许跨域请求携带 cookie
        config.setAllowCredentials(true);
        // 允许的请求方法（GET、POST、PUT 等）
        config.addAllowedMethod("*");
        // 允许的请求头（包括自定义头，如 token）
        config.addAllowedHeader("*");
        // 映射所有请求路径
        source.registerCorsConfiguration("/**", config);
        return new CorsFilter(source);
    }
}

```

### 二、核心文件代码

#### 1. src/main.js（程序入口）
```javascript
 import { createApp } from 'vue'
import App from './App.vue'
import router from './router'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'
import axios from 'axios'


createApp(App).use(router).mount('#app')
// 配置后端接口基础地址（与后端端口一致）
axios.defaults.baseURL = 'http://localhost:8083'

// 请求拦截器：携带token
axios.interceptors.request.use(config => {
    const token = localStorage.getItem('token')
    if (token) {
        config.headers.token = token
    }
    return config
})
// 响应拦截器：处理错误
axios.interceptors.response.use(
    response => response,
    error => {
        if (error.response?.status === 401) {
            // 未登录或token失效，跳转到登录页
            localStorage.removeItem('token')
            router.push('/')
            alert('登录已过期，请重新登录')
        }
        return Promise.reject(error)
    }
)

const app = createApp(App)
app.use(router)
app.use(ElementPlus)
app.config.globalProperties.$axios = axios // 全局挂载axios
app.mount('#app')
```


#### 2. src/router/index.js（路由配置）
```javascript
import { createRouter, createWebHistory } from 'vue-router'
import Login from '../views/Login.vue'
import Home from '../views/Home.vue'

const routes = [
    { path: '/', name: 'Login', component: Login },
    {
        path: '/home',
        name: 'Home',
        component: Home,
        meta: { requiresAuth: true }
    }
]

const router = createRouter({
    history: createWebHistory(),
    routes
})

// 路由守卫
router.beforeEach((to, from, next) => {
    if (to.meta.requiresAuth && !localStorage.getItem('token')) {
        next('/')
    } else {
        next()
    }
})

export default router
```


#### 3. src/views/Login.vue（登录页面）
```vue
 <template>
  <div class="login-box">
    <h2>IoT 系统登录</h2>
    <div class="form-group">
      <label>用户名：</label>
      <input v-model="username" type="text" placeholder="请输入用户名" />
    </div>
    <div class="form-group">
      <label>密码：</label>
      <input v-model="password" type="password" placeholder="请输入密码" />
    </div>
    <button @click="login">登录</button>
    <p class="error">{{ errorMsg }}</p>
    <p class="debug">调试信息：{{ debugMsg }}</p>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { useRouter } from 'vue-router'
import axios from 'axios'

const router = useRouter()
const username = ref('admin') // 测试账号
const password = ref('123456') // 测试密码
const errorMsg = ref('')
const debugMsg = ref('') // 显示调试信息

const login = async () => {
  errorMsg.value = ''
  debugMsg.value = '开始登录请求...'

  try {
    // 1. 构建表单格式参数（关键修改）
    const formData = new FormData()
    formData.append('username', username.value)
    formData.append('password', password.value)

    // 2. 发送登录请求
    const res = await axios.post(
        'http://localhost:8083/findUser',
        formData, // 使用表单数据
        {
          headers: {
            'Content-Type': 'multipart/form-data' // 声明表单格式
          }
        }
    )

    // 3. 打印完整响应（方便调试）
    debugMsg.value = `后端返回: code=${res.data.code}, 用户=${res.data.data?.users?.username || '空'}`
    console.log('完整响应:', res.data)

    // 4. 判断登录结果
    if (res.data.code === 200 && res.data.data.users) {
      localStorage.setItem('token', res.data.data.token)
      localStorage.setItem('user', JSON.stringify(res.data.data.users))
      router.push('/home') // 登录成功跳首页
    } else {
      errorMsg.value = res.data.msg || '用户名或密码错误'
    }
  } catch (err) {
    debugMsg.value = `请求失败: ${err.message}`
    errorMsg.value = '连接后端失败，请检查服务是否启动'
  }
}
</script>
```

#### 5. src/views/ResetPassword.vue（密码重置页面）
```vue
 <template>
  <div class="reset-container">
    <el-card title="密码重置" class="reset-card">
      <el-form ref="resetForm" :model="resetForm" :rules="rules" label-width="100px">
        <el-form-item label="邮箱" prop="email">
          <el-input v-model="resetForm.email" placeholder="请输入注册邮箱"></el-input>
        </el-form-item>
        <el-form-item label="验证码" prop="code">
          <el-row :gutter="10">
            <el-col :span="16">
              <el-input v-model="resetForm.code" placeholder="请输入验证码"></el-input>
            </el-col>
            <el-col :span="8">
              <el-button
                  type="primary"
                  @click="sendCode"
                  :disabled="countdown > 0"
              >
                {{ countdown > 0 ? `${countdown}秒后重发` : '发送验证码' }}
              </el-button>
            </el-col>
          </el-row>
        </el-form-item>
        <el-form-item label="新密码" prop="newPassword">
          <el-input v-model="resetForm.newPassword" type="password" placeholder="请输入新密码"></el-input>
        </el-form-item>
        <el-form-item>
          <el-button type="primary" @click="handleReset">重置密码</el-button>
          <el-button @click="$router.push('/')" style="margin-left: 10px;">返回登录</el-button>
        </el-form-item>
      </el-form>
    </el-card>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { useRouter } from 'vue-router'
import { ElMessage } from 'element-plus'

const router = useRouter()
const resetForm = ref({
  email: '',
  code: '',
  newPassword: ''
})
const countdown = ref(0) // 验证码倒计时

// 表单验证规则
const rules = ref({
  email: [
    { required: true, message: '请输入邮箱', trigger: 'blur' },
    { type: 'email', message: '请输入正确的邮箱格式', trigger: 'blur' }
  ],
  code: [
    { required: true, message: '请输入验证码', trigger: 'blur' }
  ],
  newPassword: [
    { required: true, message: '请输入新密码', trigger: 'blur' },
    { min: 6, message: '密码长度不能少于6位', trigger: 'blur' }
  ]
})

// 发送验证码
const sendCode = () => {
  if (!resetForm.value.email) {
    ElMessage.warning('请先输入邮箱')
    return
  }
  // 调用后端发送验证码接口
  $axios.post('/send-reset-code', { email: resetForm.value.email })
      .then(response => {
        if (response.data.code === 200) {
          ElMessage.success('验证码已发送至邮箱')
          // 开始倒计时
          countdown.value = 60
          const timer = setInterval(() => {
            countdown.value--
            if (countdown.value <= 0) {
              clearInterval(timer)
            }
          }, 1000)
        } else {
          ElMessage.error(response.data.msg)
        }
      })
      .catch(error => {
        ElMessage.error('发送失败，请检查邮箱是否正确')
        console.error(error)
      })
}

// 重置密码
const handleReset = () => {
  // 调用后端重置密码接口
  $axios.post('/reset-password', resetForm.value)
      .then(response => {
        if (response.data.code === 200) {
          ElMessage.success('密码重置成功，请登录')
          router.push('/') // 跳回登录页
        } else {
          ElMessage.error(response.data.msg)
        }
      })
      .catch(error => {
        ElMessage.error('重置失败，请检查信息')
        console.error(error)
      })
}
</script>

<style scoped>

  height: 100vh;
  background-color: #f5f7fa;
}
.reset-card {
  width: 500px;
  padding: 20px;
}
</style>
```
